}

\usepackage{xcolor}

\definecolor{cvNavy}{HTML}{1A365D}
\definecolor{cvTitle}{HTML}{4A5568}
\definecolor{cvMuted}{HTML}{718096}
\definecolor{cvText}{HTML}{2D3748}
\definecolor{cvRule}{HTML}{E2E8F0}
\definecolor{cvAccent}{HTML}{CBD5E1}

\color{cvText}
\setlength{\parskip}{4pt}
\setlength{\parindent}{0pt}
\linespread{1.15}

\hypersetup{
    colorlinks=true,
    linkcolor=cvNavy,
    urlcolor=cvNavy,
    pdftitle={[YOUR_NAME] - CV},
}

\setlist[itemize]{leftmargin=18pt, topsep=2pt, itemsep=3pt, parsep=0pt, label=\textbullet}

\newcommand{\cvsection}[1]{%
  \vspace{14pt}%
  {\color{cvNavy}\bfseries\fontsize{12}{14}\selectfont\MakeUppercase{#1}\par}%
  \vspace{4pt}%
  {\color{cvRule}\rule{\linewidth}{2pt}}%
  \vspace{8pt}%
}

\newcommand{\cvheader}[3]{%
  {\fontsize{28}{30}\bfseries\color{cvNavy}\MakeUppercase{#1}\par}
  \vspace{2pt}
  {\fontsize{14}{16}\bfseries\color{cvTitle}#2\par}
  \vspace{4pt}
  {\fontsize{9.5}{12}\color{cvMuted}#3\par}
  \vspace{6pt}
  {\color{cvNavy}\rule{\linewidth}{3pt}}
  \vspace{10pt}
}

\newcommand{\cvskillcategory}[2]{%
  \vspace{2pt}
  \noindent
  \raisebox{0pt}[0pt][0pt]{%
    \hspace{6pt}\textcolor{cvAccent}{\rule{3pt}{\baselineskip}}%
    \hspace{6pt}%
    \parbox[t]{0.97\linewidth}{\textbf{\color{cvNavy}#1}: #2}%
  }
  \par
}

\newcommand{\cvjob}[3]{%
  \needspace{5\baselineskip}
  {\fontsize{11}{13}\bfseries\color{cvText}#1\par}
  \vspace{1pt}
  {\fontsize{9.5}{12}\color{cvMuted}#2\par}
  \vspace{4pt}
  #3
  \vspace{8pt}
}

\newcommand{\cveducation}[2]{%
  #1\par
  \vspace{2pt}
  {\fontsize{9.5}{12}\color{cvMuted}#2\par}
  \vspace{6pt}
}

\begin{document}

\cvjob{[Job Title]}{[Company] \textbar{} [City, Country] \textbar{} [YYYY--Present]}{%
\begin{itemize}
  \item [Achievement or responsibility 1 -- be specific, use numbers where possible]
  \item [Achievement or responsibility 2]
  \item [Achievement or responsibility 3]
\end{itemize}}

\cvjob{[Job Title]}{[Company] \textbar{} [City, Country] \textbar{} [YYYY--YYYY]}{%
\begin{itemize}

  \item [Achievement or responsibility 2]
\end{itemize}}

\cvsection{Education}
\cveducation{\textbf{[Degree] in [Field]}}{[Institution] \textbar{} [City, Country] \textbar{} [YYYY--YYYY]}

\cvsection{Languages}

\end{document}
